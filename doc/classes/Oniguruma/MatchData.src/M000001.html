      def initialize( string, starts, ends, names )
         @string = string
         @starts = starts
         @ends = ends
         @matches = []
         @starts.size.times do |i|
            @matches << @string[@starts[i]...@ends[i]]
         end
         @match_count = @matches.size
         @start_pos = 0
         @names = names
      end
